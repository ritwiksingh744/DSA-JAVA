            String[] S = s.split(" ");
            int[] v = new int[2 * n + 2];
            for(int i = 0; i < 2 * n + 2; i++)
            {
                v[i] = Integer.parseInt(S[i]);
            }
            Solution ob = new Solution();

                System.out.print(ans[i] + " ");
            System.out.println();
        }
    }

class Solution
{
    public int[] singleNumber(int[] nums)
    {
        // Co
        int n=nums.length;
        int sum=0;
        for(int i=0;i<n;i++){
            sum=(sum^nums[i]);
            }
        sum=(sum & -sum);
        int sum1=0;
        int sum2=0;
        for(int i=0;i<nums.length;i++){
            if((nums[i] & sum)>0){
                sum1=(sum1^nums[i]);
                 }
            else 
                sum2=(sum2^nums[i]);
            }
        int arr[]=new int[]{sum1,sum2};
        
        Arrays.sort(arr);
        
        return arr;
        
    }
}
