}

class Solution{
    //Function to find the leaders in the array.
    static ArrayList<Integer> leaders(int arr[], int n){
        // Your code here
        ArrayList<Integer> alist =new ArrayList<>();
        int currldr=arr[n-1];
        alist.add(currldr);
        for(int i=n-2;i>=0;i--)
        {
            if(currldr<=arr[i])
            {
                alist.add(arr[i]);
                currldr=arr[i];
            }
            
        }
        Collections.reverse(alist);
            return alist;
    }
}
